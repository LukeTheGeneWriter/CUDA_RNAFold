HelloWorld
==========

Below, you'll find some more or less simple C programs showing first steps into using *RNAlib* . A complete list of example C programs can be found in the :ref:`C Examples <doxid-examples_c>` section.

.. rubric:: Simple MFE prediction for a given sequence

.. ref-code-block:: cpp

    #include <stdlib.h>
    #include <stdio.h>
    #include <string.h>
    
    #include <ViennaRNA/fold.h>
    #include <ViennaRNA/utils.h>
    
    int
    main()
    {
      /* The RNA sequence */
      char  *seq = "GAGUAGUGGAACCAGGCUAUGUUUGUGACUCGCAGACUAACA";
    
      /* allocate memory for MFE structure (length + 1) */
      char  *structure = (char *):ref:`vrna_alloc <doxid-group__utils_1gaf37a0979367c977edfb9da6614eebe99>`(sizeof(char) * (strlen(seq) + 1));
    
      /* predict Minmum Free Energy and corresponding secondary structure */
      float mfe = :ref:`vrna_fold <doxid-group__mfe__fold__single_1ga29a33b2895f4e67b0480271ff289afdc>`(seq, structure);
    
      /* print sequence, structure and MFE */
      printf("%s\n%s [ %6.2f ]\n", seq, structure, mfe);
    
      /* cleanup memory */
      free(structure);
    
      return 0;
    }

.. rubric:: Simple MFE prediction for a multiple sequence alignment

.. ref-code-block:: cpp

    #include <stdlib.h>
    #include <stdio.h>
    #include <string.h>
    
    #include <ViennaRNA/alifold.h>
    #include <ViennaRNA/aln_util.h>
    #include <ViennaRNA/utils.h>
    
    int
    main()
    {
      /* The RNA sequence alignment */
      const char  *sequences[] = {
        "CUGCCUCACAACGUUUGUGCCUCAGUUACCCGUAGAUGUAGUGAGGGU",
        "CUGCCUCACAACAUUUGUGCCUCAGUUACUCAUAGAUGUAGUGAGGGU",
        "---CUCGACACCACU---GCCUCGGUUACCCAUCGGUGCAGUGCGGGU",
        NULL /* indicates end of alignment */
      };
    
      /* compute the consensus sequence */
      char        *cons = :ref:`consensus <doxid-group__aln__utils_1ga8e8b805faaf6dee2feb4cbc38ba48e71>`(sequences);
    
      /* allocate memory for MFE consensus structure (length + 1) */
      char        *structure = (char *):ref:`vrna_alloc <doxid-group__utils_1gaf37a0979367c977edfb9da6614eebe99>`(sizeof(char) * (strlen(sequences[0]) + 1));
    
      /* predict Minmum Free Energy and corresponding secondary structure */
      float       mfe = :ref:`vrna_alifold <doxid-group__consensus__mfe__fold_1ga6c9d3bef3e92c6d423ffac9f981418c1>`(sequences, structure);
    
      /* print consensus sequence, structure and MFE */
      printf("%s\n%s [ %6.2f ]\n", cons, structure, mfe);
    
      /* cleanup memory */
      free(cons);
      free(structure);
    
      return 0;
    }

.. rubric:: Simple Base Pair Probability computation

.. ref-code-block:: cpp

    #include <stdlib.h>
    #include <stdio.h>
    #include <string.h>
    
    #include <ViennaRNA/fold.h>
    #include <ViennaRNA/part_func.h>
    #include <ViennaRNA/utils.h>
    
    int
    main()
    {
      /* The RNA sequence */
      char      *seq = "GAGUAGUGGAACCAGGCUAUGUUUGUGACUCGCAGACUAACA";
    
      /* allocate memory for pairing propensity string (length + 1) */
      char      *propensity = (char *):ref:`vrna_alloc <doxid-group__utils_1gaf37a0979367c977edfb9da6614eebe99>`(sizeof(char) * (strlen(seq) + 1));
    
      /* pointers for storing and navigating through base pair probabilities */
      :ref:`vrna_ep_t <doxid-structvrna__elem__prob__s>` *ptr, *pair_probabilities = NULL;
    
      float     en = :ref:`vrna_pf_fold <doxid-group__pf__fold_1gac4a2a74a79e49818bc35412a2b392c7e>`(seq, propensity, &pair_probabilities);
    
      /* print sequence, pairing propensity string and ensemble free energy */
      printf("%s\n%s [ %6.2f ]\n", seq, propensity, en);
    
      /* print all base pairs with probability above 50% */
      for (ptr = pair_probabilities; ptr->:ref:`i <doxid-structvrna__elem__prob__s_1a0f8bb11ded4e605f816d7ad92eb568f6>` != 0; ptr++)
        if (ptr->:ref:`p <doxid-structvrna__elem__prob__s_1a9c09385582d8a7ab00485181f4e868b7>` > 0.5)
          printf("p(%d, %d) = %g\n", ptr->:ref:`i <doxid-structvrna__elem__prob__s_1a0f8bb11ded4e605f816d7ad92eb568f6>`, ptr->:ref:`j <doxid-structvrna__elem__prob__s_1acada5be62ed6843334a918ca543f0c0d>`, ptr->:ref:`p <doxid-structvrna__elem__prob__s_1a9c09385582d8a7ab00485181f4e868b7>`);
    
      /* cleanup memory */
      free(pair_probabilities);
      free(propensity);
    
      return 0;
    }

.. rubric:: Deviating from the Default Model

.. ref-code-block:: cpp

    #include <stdlib.h>
    #include <stdio.h>
    #include <string.h>
    
    #include <ViennaRNA/mfe.h>
    #include <ViennaRNA/fold_compound.h>
    #include <ViennaRNA/model.h>
    #include <ViennaRNA/string_utils.h>
    #include <ViennaRNA/utils.h>
    
    int
    main()
    {
      /* initialize random number generator */
      :ref:`vrna_init_rand <doxid-group__utils_1ga0ad1f40ea316e5c5918695c35613027a>`();
    
      /* Generate a random sequence of 50 nucleotides */
      char      *seq = :ref:`vrna_random_string <doxid-group__string__utils_1ga4eeb3750dcf860b9f3158249f95dbd7f>`(50, "ACGU");
    
      /* allocate memory for MFE structure (length + 1) */
      char      *structure = (char *):ref:`vrna_alloc <doxid-group__utils_1gaf37a0979367c977edfb9da6614eebe99>`(sizeof(char) * (strlen(seq) + 1));
    
      /* create a new model details structure to store the Model Settings */
      :ref:`vrna_md_t <doxid-structvrna__md__s>` md;
    
      /* ALWAYS set default model settings first! */
      :ref:`vrna_md_set_default <doxid-group__model__details_1ga8ac6ff84936282436f822644bf841f66>`(&md);
    
      /* change temperature and activate G-Quadruplex prediction */
      md.:ref:`temperature <doxid-structvrna__md__s_1a5f7e5c2b65bada5188443470e576aa4b>`  = 25.0; /* 25 Deg Celcius */
      md.:ref:`gquad <doxid-structvrna__md__s_1af88a511a2b1f526b4c6213de6cb8fd6e>`        = 1;    /* Turn-on G-Quadruples support */
    
      /* create a fold compound */
      :ref:`vrna_fold_compound_t <doxid-structvrna__fc__s>`  *fc = :ref:`vrna_fold_compound <doxid-group__fold__compound_1ga6601d994ba32b11511b36f68b08403be>`(seq, &md, :ref:`VRNA_OPTION_DEFAULT <doxid-group__fold__compound_1gacea5b7ee6181c485f36e2afa0e9089e4>`);
    
      /* predict Minmum Free Energy and corresponding secondary structure */
      float                 mfe = :ref:`vrna_mfe <doxid-group__mfe__fold_1gabd3b147371ccf25c577f88bbbaf159fd>`(fc, structure);
    
      /* print sequence, structure and MFE */
      printf("%s\n%s [ %6.2f ]\n", seq, structure, mfe);
    
      /* cleanup memory */
      free(structure);
      :ref:`vrna_fold_compound_free <doxid-group__fold__compound_1gadded6039d63f5d6509836e20321534ad>`(fc);
    
      return 0;
    }
